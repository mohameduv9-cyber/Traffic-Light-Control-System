#define F_CPU 8000000UL
#include <avr/io.h>
#include <util/delay.h>
#include <stdio.h>

#define LCD_ADDR 0x4E
#define RS_BIT   0
#define EN_BIT   2
#define BL_BIT   3

#define DHT_PIN  PD2
#define SEG_PORT PORTA

#define GREEN1   PB2
#define YELLOW1  PB1
#define RED1     PB0
#define GREEN2   PB5
#define YELLOW2  PB4
#define RED2     PB3

uint8_t I_Temp = 25, I_RH = 50, msg_index = 0;

void i2c_init() {
	TWSR = 0x00;
	TWBR = 0x20;
	TWCR = (1 << TWEN);
}

void i2c_write(uint8_t data) {
	TWDR = data;
	TWCR = (1 << TWINT) | (1 << TWEN);
	while (!(TWCR & (1 << TWINT)));
}

void i2c_start() {
	TWCR = (1 << TWINT) | (1 << TWSTA) | (1 << TWEN);
	while (!(TWCR & (1 << TWINT)));
}

void i2c_stop() {
	TWCR = (1 << TWINT) | (1 << TWSTO) | (1 << TWEN);
}

void lcd_send(uint8_t val, uint8_t mode) {
	uint8_t high = (val & 0xF0) | mode | (1 << BL_BIT);
	uint8_t low = ((val << 4) & 0xF0) | mode | (1 << BL_BIT);
	
	i2c_start();
	i2c_write(LCD_ADDR);
	
	i2c_write(high | (1 << EN_BIT));
	_delay_us(1);
	i2c_write(high & ~(1 << EN_BIT));
	
	i2c_write(low | (1 << EN_BIT));
	_delay_us(1);
	i2c_write(low & ~(1 << EN_BIT));
	
	i2c_stop();
	_delay_ms(2);
}

void lcd_string_i2c(const char *s) {
	while (*s) lcd_send(*s++, 1);
}

void lcd_init_i2c() {
	_delay_ms(50);
	lcd_send(0x03, 0);
	lcd_send(0x03, 0);
	lcd_send(0x03, 0);
	lcd_send(0x02, 0);
	lcd_send(0x28, 0);
	lcd_send(0x0C, 0);
	lcd_send(0x01, 0);
}

void read_dht11() {
	uint8_t bits[5] = {0, 0, 0, 0, 0};
	uint8_t i, j;

	DDRD |= (1 << DHT_PIN);
	PORTD &= ~(1 << DHT_PIN);
	_delay_ms(18);
	
	PORTD |= (1 << DHT_PIN);
	DDRD &= ~(1 << DHT_PIN);
	_delay_us(30);

	if (!(PIND & (1 << DHT_PIN))) {
		while (!(PIND & (1 << DHT_PIN)));
		while (PIND & (1 << DHT_PIN));

		for (i = 0; i < 5; i++) {
			for (j = 0; j < 8; j++) {
				while (!(PIND & (1 << DHT_PIN)));
				_delay_us(30);
				if (PIND & (1 << DHT_PIN)) bits[i] |= (1 << (7 - j));
				while (PIND & (1 << DHT_PIN));
			}
		}

		if (bits[0] != 0 || bits[2] != 0) {
			I_RH = bits[0];
			I_Temp = bits[2];
		}
	}
}

void update_system() {
	char buf[20];
	const char* messages[] = {
		" Staying in lane    ",
		" Fasten seatbelt    ",
		" No phone driving   ",
		" Focus on leadership"
	};

	read_dht11();

	lcd_send(0x80, 0);
	sprintf(buf, "T:%uC H:%u%%      ", I_Temp, I_RH);
	lcd_string_i2c(buf);

	lcd_send(0xC0, 0);
	lcd_string_i2c(messages[msg_index]);
	msg_index = (msg_index + 1) % 4;
}

void multiplex(uint8_t n1, uint8_t n2) {
	uint8_t digits[] = {n1 % 10, n1 / 10, n2 % 10, n2 / 10};
	uint8_t pins[]   = {PA4, PA5, PA6, PA7};
	
	for (int i = 0; i < 4; i++) {
		SEG_PORT = (digits[i] & 0x0F) | (1 << pins[i]);
		_delay_ms(2);
		SEG_PORT = 0x00;
	}
}

void countDownOneSec(uint8_t t1, uint8_t t2, int blinkPin) {
	update_system();
	for (int j = 0; j < 110; j++) {
		if (blinkPin != -1 && j % 30 == 0) {
			PORTB ^= (1 << blinkPin);
		}
		multiplex(t1, t2);
	}
}

int main(void) {
	DDRB = 0xFF;
	DDRA = 0xFF;
	DDRD &= ~(1 << DHT_PIN);
	
	i2c_init();
	_delay_ms(500);
	lcd_init_i2c();

	while (1) {
		PORTB = (1 << GREEN1) | (1 << RED2);
		for (int i = 37; i > 3; i--) countDownOneSec(i, i + 3, -1);
		for (int i = 3; i > 0; i--)  countDownOneSec(i, i + 3, GREEN1);

		PORTB = (1 << YELLOW1) | (1 << RED2);
		for (int i = 3; i > 0; i--)  countDownOneSec(i, i, -1);

		PORTB = (1 << RED1) | (1 << GREEN2);
		for (int i = 37; i > 3; i--) countDownOneSec(i + 3, i, -1);
		for (int i = 3; i > 0; i--)  countDownOneSec(i + 3, i, GREEN2);

		PORTB = (1 << RED1) | (1 << YELLOW2);
		for (int i = 3; i > 0; i--)  countDownOneSec(i, i, -1);
	}
}
